Comparación de matrícula masculina y femenina (6 a 10 años)

packages = ["matplotlib", "seaborn", "pandas"] import matplotlib.pyplot as plt import seaborn as sns import pandas as pd # Ejemplo de datos ficticios df = pd.DataFrame({ 'matriculacion_masc_6a10': [120, 130, 140, 150, 160], 'matriculacion_fem_6a10': [110, 115, 120, 125, 130], }) plt.style.use('seaborn-v0_8-whitegrid') sns.set_palette("crest") plt.figure(figsize=(8,4)) df[['matriculacion_masc_6a10', 'matriculacion_fem_6a10']].plot(kind='bar') plt.title('Comparación de matrícula (6 a 10 años)') plt.xlabel('Departamento') plt.ylabel('Número de matrículas') plt.legend(['Masculina', 'Femenina']) plt.show()
